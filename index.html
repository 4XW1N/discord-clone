function promptCreateServer() { const rawOption = prompt("Type 'CREATE' to make a new server, or paste an existing Invite Code:"); // If they hit Cancel, rawOption is null. Exit safely! if (rawOption === null) return; const option = rawOption.trim(); if (!option) return; if (option.toUpperCase() === 'CREATE') { const serverName = prompt("Enter server hub name:"); if (serverName === null) return; // Safe exit on cancel const cleanName = serverName.trim(); if (!cleanName) return; appSettings.networkID = "server-" + Math.random().toString(36).substring(2, 9) + "-" + cleanName.toLowerCase().replace(/\s+/g, '-'); appSettings.groupChannels = ['general', 'gaming']; appSettings.directMessages = []; appSettings.pendingRequests = []; currentGroup = 'general'; currentDM = ''; } else { appSettings.networkID = option; appSettings.groupChannels = ['general']; appSettings.directMessages = []; appSettings.pendingRequests = []; currentGroup = 'general'; currentDM = ''; } localStorage.setItem('discord_clone_user', JSON.stringify(appSettings)); loadCacheFromLocalStorage(); completeStartup(); toggleMobileNav(false); }
